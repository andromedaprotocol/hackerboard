    // Sidebar Toggle
    document.addEventListener('DOMContentLoaded', function () {
      const sidebarToggle = document.getElementById('sidebarToggle');
      const sidebar = document.querySelector('.sidebar');
      const mainContent = document.querySelector('.main-content');

      function toggleSidebar() {
        sidebar.classList.toggle('collapsed');
        mainContent.classList.toggle('expanded');
      }

      sidebarToggle.addEventListener('click', toggleSidebar);

      if (window.innerWidth <= 768) {
        toggleSidebar();
      }

      window.addEventListener('resize', function () {
        if (window.innerWidth <= 768 && !sidebar.classList.contains('collapsed')) {
          toggleSidebar();
        }
      });
    });

    // Client-Side Validation & Form Submission
    document.addEventListener('DOMContentLoaded', function () {
      // Basic validation rules
      const validationRules = {
        name: {
          validate: value => value.trim().length >= 2,
          message: 'Please enter your full name (minimum 2 characters)'
        },
        email: {
          validate: value => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value),
          message: 'Please enter a valid email address'
        },
        github: {
          validate: value =>
            value.startsWith('https://github.com/') ||
            value.startsWith('https://www.github.com/'),
          message: 'Please enter a valid GitHub profile URL (https://github.com/...)'
        },
        telegram: {
          validate: value => value.trim().length > 0,
          message: 'Please enter your Telegram username'
        },
        location: {
          validate: value => value.trim().length > 0,
          message: 'Please enter your location'
        },
        x: {
          // Not strictly required, but let's provide a placeholder rule
          validate: value => true, // optional
          message: ''
        },
        portfolio: {
          validate: value =>
            !value.trim() || value.startsWith('http://') || value.startsWith('https://'),
          message: 'Please enter a valid portfolio URL (or leave empty)'
        }
      };

      // Attach validation checks to fields
      Object.keys(validationRules).forEach(fieldId => {
        const input = document.getElementById(fieldId);
        if (input) {
          input.addEventListener('input', () => validateField(input));
          input.addEventListener('blur', () => validateField(input));
        }
      });

      function validateField(input) {
        const formGroup = input.closest('.form-group');
        const rule = validationRules[input.id];
        if (!rule) return true;

        const errorMessage = formGroup.querySelector('.error-message');
        const isValid = rule.validate(input.value);

        formGroup.classList.remove('error', 'success');
        if (!isValid) {
          formGroup.classList.add('error');
          errorMessage.textContent = rule.message;
        } else {
          formGroup.classList.add('success');
        }
        return isValid;
      }

      // Auto-format GitHub URL
      document.getElementById('github').addEventListener('blur', function () {
        let value = this.value.trim();
        if (!value.includes('github.com') && !value.includes('http')) {
          value = `https://github.com/${value.replace('@', '')}`;
        }
        if (value.startsWith('http://')) {
          value = value.replace('http://', 'https://');
        }
        if (!value.startsWith('http')) {
          value = `https://${value}`;
        }
        this.value = value;
        validateField(this);
      });

      // Auto-format Telegram handle
      document.getElementById('telegram').addEventListener('blur', function () {
        let value = this.value.trim();
        if (value && !value.startsWith('@')) {
          this.value = '@' + value;
        }
      });

      // Auto-format X handle
      document.getElementById('x').addEventListener('blur', function () {
        let value = this.value.trim();
        if (value && !value.startsWith('@')) {
          this.value = '@' + value;
        }
      });

      // Auto-format portfolio
      document.getElementById('portfolio').addEventListener('focus', function () {
        if (this.value === '') {
          this.value = 'https://';
        }
      });
      document.getElementById('portfolio').addEventListener('blur', function () {
        if (this.value === 'https://') {
          this.value = '';
        }
      });

      // Attempt simple IP-based location
      async function detectUserLocation() {
        try {
          const response = await fetch('https://ipapi.co/json/');
          const data = await response.json();
          const locationInput = document.getElementById('location');
          if (data.city && data.country_name) {
            locationInput.value = `${data.city}, ${data.country_name}`;
            const event = new Event('input');
            locationInput.dispatchEvent(event);
          }
        } catch (error) {
          console.log('Location detection failed:', error);
        }
      }
      detectUserLocation();

      // Form submission
      document.getElementById('onboardingForm').addEventListener('submit', async function (e) {
        e.preventDefault();

        // Validate all fields before submission
        let isValid = true;
        Object.keys(validationRules).forEach(fieldId => {
          const input = document.getElementById(fieldId);
          if (input && !validateField(input)) {
            isValid = false;
          }
        });
        if (!isValid) {
          alert('Please correct the errors before submitting.');
          return;
        }

        const submitButton = e.target.querySelector('button[type="submit"]');
        const originalText = submitButton.innerHTML;
        submitButton.disabled = true;
        submitButton.innerHTML = '<span class="loading-spinner"></span>Submitting...';

        // Gather form data
        const data = {

          const form = document.getElementById('onboardingForm');

          // Clear form
          form.reset();

          console.error('Error:', error);
          alert('There was an error submitting the form. Please try again.');

          submitButton.disabled = false;
          submitButton.innerHTML = originalText;
        }

        const recDiv = document.getElementById("recommendations");
        const recList = document.getElementById("recommendation-list");
        recList.innerHTML = '';
        if (recommendations && recommendations.length > 0) {

            recList.appendChild(li);
          });
          recDiv.style.display = 'block';
        } else {
          recDiv.style.display = 'none';
        }
      }
    });
